(1) simple if statement
            if(condition){
                statements;
            }

        (2) if...else statement
            if(condition){
                statements
            }else{
                statements
            }

        (3) if...else if... else statement
            if(condition){
                statements
            }else if(condition){
                statements
            }else{
                statements
            }

        (4) nested if statement
            if(condition){
                if(condition){
                    statements
                }
            }

        (5) nested if... else statement
            if(condition){
                if(condition){
                    statements
                }else{
                    statements
                }
            }else{
                if(condition){
                    statements
                }else{
                    statements
                }
            }
